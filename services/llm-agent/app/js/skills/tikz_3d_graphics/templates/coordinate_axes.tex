% TikZ 3D: annotated coordinate axes with a point in spherical coords
\documentclass{article}
\usepackage{tikz}
\usepackage{tikz-3dplot}

\begin{document}
\tdplotsetmaincoords{65}{120}

\begin{tikzpicture}[tdplot_main_coords, scale=3, line width=0.7pt, font=\small]

  % Axes (draw negative halves dashed for depth cue)
  \draw[dashed, gray!60] (-0.5,0,0) -- (0,0,0);
  \draw[dashed, gray!60] (0,-0.5,0) -- (0,0,0);
  \draw[dashed, gray!60] (0,0,-0.3) -- (0,0,0);

  \draw[thick,-{Stealth}] (0,0,0) -- (1.3,0,0) node[anchor=north]{$x$};
  \draw[thick,-{Stealth}] (0,0,0) -- (0,1.3,0) node[anchor=west] {$y$};
  \draw[thick,-{Stealth}] (0,0,0) -- (0,0,1.3) node[anchor=south]{$z$};

  % Point P at r=1, polar θ=55°, azimuthal φ=35°
  \tdplotsetcoord{P}{1}{55}{35}

  % Vector OP
  \draw[-{Stealth}, red, thick] (0,0,0) -- (P)
    node[above, xshift=3pt]{$P$};

  % Dashed projections
  \draw[dashed, blue!60] (Pxy) -- (P)   node[midway, right, blue!70]{\scriptsize $z_P$};
  \draw[dashed, blue!60] (0,0,0) -- (Pxy);
  \draw[dashed, blue!60] (Px)  -- (Pxy) node[midway, below, blue!70]{\scriptsize $y_P$};
  \draw[dashed, blue!60] (Py)  -- (Pxy) node[midway, right, blue!70]{\scriptsize $x_P$};

  % Azimuthal arc φ
  \tdplotdrawarc[blue]{(0,0,0)}{0.30}{0}{35}{anchor=north}{$\varphi$}

  % Polar arc θ (in the rotated plane containing OP)
  \tdplotsetthetaplanecoords{35}
  \tdplotdrawarc[tdplot_rotated_coords, red]{(0,0,0)}{0.30}{0}{55}
    {anchor=south west}{$\theta$}

  % Origin dot
  \fill (0,0,0) circle (0.8pt) node[anchor=north east]{\scriptsize $O$};

\end{tikzpicture}
\end{document}
