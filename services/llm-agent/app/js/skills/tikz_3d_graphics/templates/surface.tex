% TikZ 3D: a simple saddle surface drawn as a grid of patches
% Technique: sample (u,v) grid, draw quads with light fill for depth cue
\documentclass{article}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\usepackage{xfp}  % for fpeval

\begin{document}
\tdplotsetmaincoords{65}{130}

\begin{tikzpicture}[tdplot_main_coords, scale=1.1, line width=0.3pt]

  % Draw the surface z = x^2 - y^2 sampled on a 6x6 grid over [-1,1]^2
  % Each cell is a filled quadrilateral
  \foreach \i in {-3,-2,-1,0,1,2} {
    \foreach \j in {-3,-2,-1,0,1,2} {
      % four corners of the quad (step 0.5)
      \pgfmathsetmacro{\xa}{\i*0.5}
      \pgfmathsetmacro{\xb}{(\i+1)*0.5}
      \pgfmathsetmacro{\ya}{\j*0.5}
      \pgfmathsetmacro{\yb}{(\j+1)*0.5}
      \pgfmathsetmacro{\zaa}{\xa*\xa - \ya*\ya}
      \pgfmathsetmacro{\zba}{\xb*\xb - \ya*\ya}
      \pgfmathsetmacro{\zbb}{\xb*\xb - \yb*\yb}
      \pgfmathsetmacro{\zab}{\xa*\xa - \yb*\yb}
      % color by z value (mid-cell)
      \pgfmathsetmacro{\zmid}{((\xa+\xb)/2)*((\xa+\xb)/2) - ((\ya+\yb)/2)*((\ya+\yb)/2)}
      \pgfmathsetmacro{\col}{50 + \zmid*30}  % map z in [-1,1] to [20,80]
      \filldraw[fill=blue!\col!white, draw=gray!60]
        (\xa,\ya,\zaa) -- (\xb,\ya,\zba)
        -- (\xb,\yb,\zbb) -- (\xa,\yb,\zab) -- cycle;
    }
  }

  % Axes on top
  \draw[thick,-{Stealth}] (-1.7,0,0) -- (1.7,0,0) node[right]{$x$};
  \draw[thick,-{Stealth}] (0,-1.7,0) -- (0,1.7,0) node[right]{$y$};
  \draw[thick,-{Stealth}] (0,0,-1.2) -- (0,0,1.2) node[above]{$z$};

\end{tikzpicture}

\medskip
\textit{Saddle surface $z = x^2 - y^2$ rendered as a TikZ patch grid.}
\end{document}
